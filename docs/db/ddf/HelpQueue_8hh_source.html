205  void help_enqueue (int state_idx, int helper_phase) {
206 #ifdef TEL_LOGGING
207  LOG_S(INFO)
208  << "Thread '" << current_thread_id << "': Starting to help Thread '" << state_idx << "' with by having phase = "
209  << helper_phase << '\n';
210 #endif
211  while (is_pending(state_idx, helper_phase)) {
212 
213  auto *tail_ptr = m_tail.load();
214  auto &tail = *tail_ptr;
215  auto *next_ptr = tail_ptr->next().load();
216 
217  if (tail_ptr != m_tail.load()) {
218 #ifdef TEL_LOGGING
219  LOG_S(INFO) << "Thread '" << current_thread_id << "': Tail pointer modified. Retrying ...\n";
220 #endif
221  continue;
222  }
223 
224  if (next_ptr != nullptr) {
225 #ifdef TEL_LOGGING
226  LOG_S(INFO) << "Thread '" << current_thread_id << "': Tail pointer outdated. Retrying ...\n";
227 #endif
228  help_finish_enqueue();
229  continue;
230  }
231 
232  if (!is_pending(state_idx, helper_phase)) {
233 #ifdef TEL_LOGGING
234  LOG_S(INFO) << "Thread '" << current_thread_id << "': Operation already executed. Done.\n";
235 #endif
236  return;
237  }
238 
239  auto *state_ptr = m_states.at(state_idx).load();
240  auto state = *state_ptr;
241  if (!state.pending()) {
242 #ifdef TEL_LOGGING
243  LOG_S(INFO) << "Thread '" << current_thread_id << "': Operation already executed. Done.\n";
244 #endif
245  return;
246  }
247 
248  auto *new_next_ptr = state_ptr->node();
249  if (tail.next().compare_exchange_strong(next_ptr, new_next_ptr)) {
250 #ifdef TEL_LOGGING
251  LOG_S(INFO) << "Thread '" << current_thread_id << "' in helping Thread '" << state_idx
252  << "': CAS on tail, next and node.\n";
253 #endif
254  return help_finish_enqueue();
255  }
256  }

259  void help_others (int helper_phase) {
260 #ifdef TEL_LOGGING
261  LOG_S(INFO)
262  << "Thread '" << current_thread_id << "': Helping others with helper phase = " << helper_phase << '\n';
263 #endif
264  int i = 0;
265  for (auto &atomic_state : m_states) {
266  auto state = atomic_state.load();
267  if (state->pending() && state->phase() <= helper_phase) {
268  if (state->operation() == Operation::enqueue) {
269 #ifdef TEL_LOGGING
270  LOG_S(INFO)
271  << "Thread '" << current_thread_id << "': Found operation which needs help - Thread '" << i
272  << "' which is performing push_back with phase = " << helper_phase << '\n';
273 #endif
274  help_enqueue(i, helper_phase);
275  }
276  }
277  ++i;
278  }
